38   /*  High-priority  service  */
39   void  interrupt  high_isr(void){
40        /*Timer  0  --  Display  update.  (approx.  300  us  cycle)
41         300  us  x  8  rows  =  2.4  ms  to  update  whole  display
42          which  is  around  420  Hz  refresh  rate*/
43        if  ((TMR0IE)  &&  (TMR0IF))  {
44            ShiftOutNextRow();
45  
46            if((display.activeRow++)  ==  8)  display.activeRow  =  0;
47           
48            TMR0L  =  50;  //  Reset  the  timer0  counter
49            TMR0IF  =  0;  //  Clear  the  timer0  interrupt  flag
50        } 
51   }
52  
53   /*  Low-priority  interrupt  routine  */
54   void  low_priority  interrupt  low_isr(void){
55  
56        if  ((TMR3IE)  &&  (TMR3IF))  {
57            milliseconds++;
58       
59            WRITETIMER3(53717);
60  
61            TMR3IF  =  CLEAR;  //  Clear  the  interrupt  flag
62        }
63   }
